# 用户端源码

```
src/
├── app.config.ts        # 全局配置
├── app.tsx              # 入口文件
├── pages/               # 页面
│   ├── index/          # 首页
│   ├── merchant/       # 商家详情
│   ├── cart/           # 购物车
│   ├── order/          # 订单
│   └── profile/        # 个人中心
├── components/         # 组件
│   ├── common/        # 通用组件
│   ├── business/      # 业务组件
│   └── layout/        # 布局组件
├── store/              # 状态管理
├── api/                # 接口调用
├── utils/              # 工具函数
├── styles/             # 样式
└── types/              # 类型定义
```
